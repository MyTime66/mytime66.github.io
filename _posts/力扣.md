   class Solution {
       public int largestRectangleArea(int[] heights) {
           int max = 0;
           int count = 1;
           for (int i = 0; i < heights.length; i++) {
               int begin = i - 1;
               while (begin >= 0) {
                   if (heights[begin--] >= heights[i]) count++;
                   else break;
               }
               begin = i + 1;
               while (begin < heights.length) {
                   if (heights[begin++] >= heights[i]) {
                       count++;
                   } else break;
               }
               begin = count * heights[i];
               if (begin > max) max = begin;
               count = 1;
           }
           return max;
       }
   }

   class Solution {
     public List<Integer> topKFrequent(int[] nums, int k) {
       // build hash map : character and how often it appears
       HashMap<Integer, Integer> count = new HashMap();
       for (int n: nums) {
         count.put(n, count.getOrDefault(n, 0) + 1);
       }
   
       // init heap 'the less frequent element first'
       PriorityQueue<Integer> heap =
               new PriorityQueue<Integer>((n1, n2) -> count.get(n1) - count.get(n2));
   
       // keep k top frequent elements in the heap
       for (int n: count.keySet()) {
         heap.add(n);
         if (heap.size() > k)
           heap.poll();
       }
   
       // build output list
       List<Integer> top_k = new LinkedList();
       while (!heap.isEmpty())
         top_k.add(heap.poll());
       Collections.reverse(top_k);
       return top_k;
     }
   }

   import java.util.*;
   class Solution {
      
       public static List<Integer> topKFrequent(int[] nums, int k) {
           List<Integer> list = new ArrayList<>(k + 1);
           Map<Integer, Integer> hm = new HashMap<>();
           for (int num : nums) {
               if (hm.get(num) == null) {
                   hm.put(num, 1);
               } else {
                   hm.put(num, hm.get(num) + 1);
               }
           }
           Collection<Integer> values = hm.values();
           int[] ints = new int[k];
           int count = 0;
           for (Integer value : values) {
               if (count < k) {
                   ints[count] = value;
                   count++;
                   if (count == k) {
                       for (int i = count/2 - 1; i >=0; i--) {
                           getMin(ints, i);
                       }
   
                   }
               }
   
               else {
                   if (value > ints[0]) {
                       ints[0] = value;
                       getMin(ints, 0);
                   }
               }
           }
           for (Map.Entry<Integer, Integer> entry : hm.entrySet()) {
               if (entry.getValue() >= ints[0]) {
                   list.add(entry.getKey());
                   if (list.size() == k) {
                       return list;
                   }
               }
           }
   
           return list;
       }
   
       private static void getMin(int[] list, int num) {
           int min = num;
           int left = num * 2 + 1;
           if (left < list.length) {
               if (list[min] > list[left]) {
                   min = left;
               }
               int right = num * 2 + 2;
               if (right < list.length) {
                   if (list[min] > list[right]) {
                   min = right;
               }
               }
           }
           if (num != min) {
               int middle = list[num];
               list[num] = list[min];
               list[min] = middle;
               getMin(list, min);
           }
   
       }
   }
